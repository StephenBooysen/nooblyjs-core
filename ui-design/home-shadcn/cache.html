💻 Code Example

// Initialize caching service const caching = require('./src/caching/index'); const express = require('express'); const EventEmitter = require('events'); const app = express(); const eventEmitter = new EventEmitter(); // Create cache instance with memory provider const cache = caching('memory', { 'express-app': app }, eventEmitter); // Save a value await cache.save('user:123', { name: 'John', email: 'john@example.com' }); // Retrieve a value const user = await cache.recall('user:123'); // Delete a value await cache.delete('user:123');
