💻 Code Example

// Initialize notifying service const notifying = require('./src/notifying/index'); const express = require('express'); const EventEmitter = require('events'); const app = express(); const eventEmitter = new EventEmitter(); // Create notification instance const notify = notifying('notifying', { 'express-app': app }, eventEmitter); // Subscribe to a channel notify.subscribe('user-updates', (message) => { console.log('Received message:', message); }); // Publish a message await notify.publish('user-updates', { type: 'user_login', userId: '123', timestamp: new Date().toISOString() }); // Unsubscribe from channel notify.unsubscribe('user-updates');
