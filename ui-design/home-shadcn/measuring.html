💻 Code Example

// Initialize measuring service const measuring = require('./src/measuring/index'); const express = require('express'); const EventEmitter = require('events'); const app = express(); const eventEmitter = new EventEmitter(); // Create metrics instance const metrics = measuring('measuring', { 'express-app': app }, eventEmitter); // Record a counter metric await metrics.increment('api_requests'); // Record a gauge metric await metrics.gauge('memory_usage', 1024); // Record timing metric const start = Date.now(); // ... do some work ... await metrics.timing('operation_duration', Date.now() - start); // Get metric values const requestCount = await metrics.get('api_requests');
